tASK 1:

    GET OwnsAnIPhone[] FROM USER
    SET total = 0
    LOOP FOR EACH ITEM IN OwnsAnIPhone[]
        IF CURRENT ITEM IN OwnsAnIphone[] = "Yes"
            SET total TO total + 1
        END IF
    END LOOP
    DISPLAY total
